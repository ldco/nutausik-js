export function runMigrations(db: Database.Database, fromVersion: number, toVersion: number): void {
  for (let v = fromVersion + 1; v <= toVersion; v++) {
    const fn = MIGRATIONS[v]
    if (fn) {
      db.transaction(() => fn(db))()
    }
  }
}
